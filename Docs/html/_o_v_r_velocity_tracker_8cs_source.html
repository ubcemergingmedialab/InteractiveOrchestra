178 
179  float currRelativeTimeInPrepBeat = Mathf.Round((Time.time - startTime) * 1000.0f) / 1000.0f;

181  float distanceCoveredSofar = GetDistanceCoveredSoFar(device);
182 
183  if (SizeOfSamplesList == 0 || distanceCoveredSofar != 0 )
184  {

186  Vector3 controllerPosition = batonObject.transform.position;
187 
188 
189  float controllerAcceleration = OVRInput.GetLocalControllerAcceleration(device).magnitude;
190 
191  // =========================
192  // -- Checks for the precise instance where the current controller y velocity is positive
193  // -- and the previous controller y velocity is negative.
194  // -- This is the first slope of the prep beat, so we spawn the plane here.
195  // =========================
196  if (previousYVelocity < 0 && controllerVelocity.y > 0 && !planeHasBeenSpawned)
197  {
198  prevCollisionTime = currOverallTime;
199  basePlaneCollisionPoint = controllerPosition;
200 
201  // ========================
202  // -- This is to account for controller weirdness. Sometimes although
203  // -- the velocity of the current controller is positive, it doesn't mean that it's at a higher position
204  // -- then the previous velocity. So we pick the smallest one.
205  // ========================
206  if (previousBatonPosition.y > conductorBaton.position.y)
207  {
208  horizontalPlane.SpawnPlane(conductorBaton.position);
209  BP1 = controllerPosition;
210  }
211  else
212  {
213  horizontalPlane.SpawnPlane(previousBatonPosition);
214  BP1 = previousControllerPosition;
215  }
216  planeHasBeenSpawned = true;
217  }
218 
219  float angleToBP1 = GetAngleToFirstCollisionWithBasePlane(BP1,controllerPosition);
220  float totalDistanceCoveredSoFar = distanceCoveredSofar;
221  if (BP1 == Vector3.zero) totalDistanceCoveredSoFar = 0;
222 
223  // =========================
224  // -- Three conditions under which we will add a data sample
225  // -- 1. The controller is located above the BP1 position
226  // -- 2. The BP1 has been defined
227  // -- 3. Controller y velocity is positive
228  // -- These combinations of conditions allow for data points to only be recorded for the prep gesture
229  // =========================

241  angleToBP1, // Angle to BP1. TODO find angle
242  totalDistanceCoveredSoFar, // Total distance covered so far
243  currentGestureSize, // Gesture size currently measuring.
244  currentBPMToRecord, // Current BPM being collected
245  currentTrial
246  );
247  if (planeHasBeenSpawned) BPMPred.RecordConductorSample(newConductorSample, tempoController);

346  private float GetAngleToFirstCollisionWithBasePlane(Vector3 BP1, Vector3 currentPosition)
347  {
348  if (BP1 == Vector3.zero) return -999;
349  Vector3 BP1ToCurrentPosition = (currentPosition - BP1).normalized;
350  Vector3 BP1ProjectedVectorTowardsCurrentPosition = (new Vector3(BP1ToCurrentPosition.x, 0,BP1ToCurrentPosition.z)).normalized;
351  float angle = (float)(Math.Acos(Vector3.Dot(BP1ProjectedVectorTowardsCurrentPosition, BP1ToCurrentPosition)) * (180 / (Math.PI)));
352  if (angle == 90) return -999;
353  return angle;
354  }
